## 📁 1. CREATE MOCK DECISION ENGINE

### `/engine/mockDecisionEngine.ts`

```ts
import { DecisionResult, DecisionScore } from "../types/decision";
import { RelationshipProfile } from "../types/relationship";
import { BudgetRange } from "../types/common";

interface DecisionInput {
  relationship: RelationshipProfile;
  occasion: string;
  budget: BudgetRange;
}

export function runMockDecisionEngine(
  input: DecisionInput
): DecisionResult {
  const { relationship, budget } = input;

  let safeScore = 60;
  let emotionalScore = 60;
  let boldScore = 60;

  // Relationship influence
  if (relationship.type === "partner") {
    emotionalScore += 20;
    boldScore += 10;
  }

  if (relationship.type === "colleague") {
    safeScore += 20;
    emotionalScore -= 20;
    boldScore -= 30;
  }

  // Closeness influence
  emotionalScore += relationship.closeness * 5;
  boldScore += relationship.closeness * 3;

  // Surprise tolerance
  if (relationship.surpriseTolerance === "low") {
    boldScore -= 40;
  }

  // Budget influence
  if (budget === "under_50") {
    boldScore -= 20;
  }

  const scores: DecisionScore[] = [
    {
      direction: "safe",
      score: clamp(safeScore),
      risk: "low",
      recommended: false,
    },
    {
      direction: "emotional",
      score: clamp(emotionalScore),
      risk: "medium",
      recommended: false,
    },
    {
      direction: "bold",
      score: clamp(boldScore),
      risk: "high",
      recommended: false,
    },
  ];

  const maxScore = Math.max(...scores.map((s) => s.score));
  scores.forEach((s) => (s.recommended = s.score === maxScore));

  return {
    scores,
    explanationByDirection: {
      safe: {
        whyThisWorks:
          "This option minimizes risk and ensures appreciation.",
        risks:
          "May feel predictable if emotional depth is expected.",
        emotionalSignal: "Reliability & stability",
      },
      emotional: {
        whyThisWorks:
          "Balances emotional meaning with thoughtful intent.",
        risks:
          "Requires understanding of personal preferences.",
        emotionalSignal: "Connection & care",
      },
      bold: {
        whyThisWorks:
          "Makes a strong statement and creates memorability.",
        risks:
          "High emotional risk if preferences are misjudged.",
        emotionalSignal: "Intensity & confidence",
      },
    },
  };
}

function clamp(value: number) {
  return Math.max(0, Math.min(100, value));
}
```
