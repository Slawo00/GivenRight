## 2️⃣ Replit – Fetch Pattern (einmalig, wiederverwendbar)

```ts
// lib/loadOptions.ts
import { supabase } from './supabase';

export async function loadScreen1Options() {
  const [
    relationships,
    closeness,
    occasions,
    importance
  ] = await Promise.all([
    supabase.from('q_relationship_types').select('code,label,description').order('sort_order'),
    supabase.from('q_closeness_levels').select('code,label').order('emotional_expectation'),
    supabase.from('q_occasion_types').select('code,label').order('sort_order'),
    supabase.from('q_occasion_importance_levels').select('code,label').order('importance_level')
  ]);

  if (relationships.error || closeness.error || occasions.error || importance.error) {
    throw new Error('Failed to load Screen 1 options');
  }

  return {
    relationships: relationships.data,
    closeness: closeness.data,
    occasions: occasions.data,
    importance: importance.data
  };
}
```

## 4️⃣ Persistenz (noch **keine** Entscheidung!)

```ts
// services/saveRelationship.ts
import { supabase } from './supabase';

export async function saveRelationship(userId: number, data: any) {
  const { error } = await supabase
    .from('relationships')
    .insert({
      user_id: userId,
      relationship_type: data.relationship_type,
      closeness_level: data.closeness_level,
      occasion_type: data.occasion_type,
      occasion_importance: data.occasion_importance
    });

  if (error) throw error;
}
```
