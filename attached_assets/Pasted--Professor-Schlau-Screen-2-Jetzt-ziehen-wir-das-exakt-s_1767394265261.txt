## 2️⃣ Replit – Fetch Pattern (Screen 2)

```ts
// lib/loadScreen2Options.ts
import { supabase } from './supabase';

export async function loadScreen2Options() {
  const [traits, tolerance] = await Promise.all([
    supabase
      .from('q_personality_traits')
      .select('code,label')
      .order('label'),

    supabase
      .from('q_surprise_tolerance_levels')
      .select('code,label')
      .order('tolerance_level', { ascending: true })
  ]);

  if (traits.error || tolerance.error) {
    throw new Error('Failed to load Screen 2 options');
  }

  return {
    personalityTraits: traits.data,
    surpriseTolerance: tolerance.data
  };
}
```

## 5️⃣ Persistenz (`recipients`)

> Screen 2 **legt keinen neuen Datensatz an**,
> sondern **ergänzt** den bestehenden Recipient.

```ts
// services/saveRecipientPersonality.ts
import { supabase } from './supabase';

export async function updateRecipientPersonality(
  recipientId: string,
  data: any
) {
  const { error } = await supabase
    .from('recipients')
    .update({
      personality_traits: data.personality_traits,
      surprise_tolerance: data.surprise_tolerance
    })
    .eq('recipient_id', recipientId);

  if (error) throw error;
}
```
